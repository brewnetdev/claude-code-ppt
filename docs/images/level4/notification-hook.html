mac 알림 macOS 데스크톱 알림 
# .claude/hooks/notify-mac.sh
payload=$(cat)
msg=$(echo "$payload" | jq -r '.message // "Claude needs input"' | sed 's/"/\\"/g')
title=$(echo "$payload" | jq -r '.title // "Claude Code"' | sed 's/"/\\"/g')
type=$(echo "$payload" | jq -r '.notification_type // "info"')
# notification_type별 사운드 분기 (선택)
case "$type" in
"permission_prompt") sound="Glass" ;;
"idle_prompt") sound="Submarine" ;;
"auth_success") sound="Hero" ;;
*) sound="Tink" ;;
esac
osascript -e "display notification \"$msg\" with title \"$title\" sound name \"$sound\""
exit 0
slack 알림 Slack Webhook 알림 
# .claude/hooks/notify-slack.sh
payload=$(cat)
# Webhook URL 미설정 시 silent exit
[ -z "$SLACK_WEBHOOK_URL" ] && exit 0
msg=$(echo "$payload" | jq -r '.message // "Done"')
# jq로 JSON 페이로드 안전 빌드 (escape 자동)
json=$(jq -n --arg text "[Claude] $msg" '{text: $text}')
curl -s -X POST "$SLACK_WEBHOOK_URL" -d "$json" -H "Content-Type: application/json" > /dev/null || true
exit 0
